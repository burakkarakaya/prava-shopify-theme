media ile benzer data ile reactda yapmıştım bunun snippet karşılığı. örneğin; picture, video, iframe gibi tek başına kullanabileceğim kucuk snippitlara bolebilirsin. yani Media diye bir snippet gonderip sonra onun içerisinden tipe gore Picture, Video, iframe gibi farklı snippetlrada gonderebilirsin.

import { IImage, IImageOptions } from "@/interfaces";

import { cn } from "@/helpers";

import { processImageUrl, isGifUrl } from "@/helpers";
import { cdn } from "nextemos";
// config
import appConfig from "@/app-config";
import { useInView } from "react-intersection-observer";

interface IProps {
  data: IImage;
  options?: IImageOptions;
}

const Picture: React.FC<IProps> = ({ data, options = {} }) => {
  const { ref, inView } = useInView({
    rootMargin: "200px",
    triggerOnce: true,
  });

  const {
    isLazy = true,
    WrapperClass = "",
    MediaClass = "w-full h-full object-cover",
    Title: optionsTitle = "",
    ImageSize,
    LazyImage,
  } = options;

  const cdnPath = cdn.getFile("/");
  const lazyClass = isLazy ? "lazy" : "eager";
  const title = data.Title || optionsTitle || "";
  const className = data.ClassName || "";

  let desktopImage =
    data.DesktopImage && processImageUrl(data.DesktopImage, cdnPath);
  let mobileImage =
    data.MobileImage && processImageUrl(data.MobileImage, cdnPath);
  let mobileRetina =
    data.MobileRetina && processImageUrl(data.MobileRetina, cdnPath);

  if (ImageSize) {
    if (ImageSize.desktop && desktopImage && !isGifUrl(desktopImage)) {
      desktopImage =
        data.DesktopImage &&
        cdn.thumbImage({
          path: data.DesktopImage.trim(),
          ...(typeof ImageSize.desktop === "number"
            ? { width: ImageSize.desktop }
            : {
                width: ImageSize.desktop.width || 0,
                height: ImageSize.desktop.height,
              }),
        });
    }
    if (ImageSize.mobile && mobileImage && !isGifUrl(mobileImage)) {
      mobileImage =
        data.MobileImage &&
        cdn.thumbImage({
          path: data.MobileImage.trim(),
          ...(typeof ImageSize.mobile === "number"
            ? { width: ImageSize.mobile }
            : {
                width: ImageSize.mobile.width || 0,
                height: ImageSize.mobile.height,
              }),
        });
    }
    if (ImageSize.mobile2x && mobileRetina && !isGifUrl(mobileRetina)) {
      mobileRetina =
        data.MobileRetina &&
        cdn.thumbImage({
          path: data.MobileRetina.trim(),
          ...(typeof ImageSize.mobile2x === "number"
            ? { width: ImageSize.mobile2x }
            : {
                width: ImageSize.mobile2x.width || 0,
                height: ImageSize.mobile2x.height,
              }),
        });
    }
  }
  const loadingImg = LazyImage
    ? LazyImage
    : "data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAAAAUAAAAGCAQAAACh8pzAAAAAEElEQVR42mN8V88ABYzkMQE/CwiVDB4roAAAAABJRU5ErkJggg==";

  return (
    <div
      ref={ref}
      className={cn("picture prm-media", lazyClass, WrapperClass, className)}
    >
      <picture>
        {isLazy && (
          <source
            media={inView ? "(max-width:0px)" : "(max-width:9999px)"}
            srcSet={loadingImg}
            className="flex items-center justify-center"
          />
        )}
        {desktopImage &&
          desktopImage !== "" &&
          mobileImage &&
          mobileImage !== "" && (
            <>
              <source
                media={`(max-width: ${appConfig.MAX_MATCHES})`}
                srcSet={`${mobileImage} 1x, ${mobileRetina} 2x`}
              />
              <source
                media={`(min-width: ${appConfig.MIN_MATCHES})`}
                srcSet={desktopImage}
              />
            </>
          )}
        <img className={MediaClass} alt={title} src={desktopImage} />
      </picture>
    </div>
  );
};

export default Picture;
